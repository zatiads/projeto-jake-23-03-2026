    43→    def decorated(*args, **kwargs):
    44→        if not session.get("logged_in"):
    45→            return redirect(url_for("login_page"))
    46→        return f(*args, **kwargs)
    47→    return decorated

   121→def _send_telegram(text):
   122→    token   = os.environ.get("TELEGRAM_BOT_TOKEN",    "").strip()
   123→    chat_id = (os.environ.get("TELEGRAM_ALERT_CHAT_ID", "").strip()
   124→               or os.environ.get("AUTHORIZED_ID", "").strip())
   125→    if not token or not chat_id:
   126→        return False, "Telegram não configurado"
   127→    try:
   128→        r = requests.post(
   129→            f"https://api.telegram.org/bot{token}/sendMessage",
   130→            json={"chat_id": chat_id, "text": text, "disable_web_page_preview": True},
   131→            timeout=10,
   132→        )
   133→        return r.ok, "Enviado" if r.ok else r.text
   134→    except Exception as e:
   135→        return False, str(e)

   158→def _chat_with_tools(client, messages):
   159→    response = client.chat.completions.create(
   160→        model="gpt-4o",
   161→        messages=messages,
   162→        tools=[TELEGRAM_TOOL],
   163→        tool_choice="auto",
   164→    )
   165→    msg = response.choices[0].message
   166→    if getattr(msg, "tool_calls", None):
   167→        for tc in msg.tool_calls:
   168→            if getattr(tc.function, "name", "") == "send_telegram_message":
   169→                args = json.loads(getattr(tc.function, "arguments", "{}") or "{}")
   170→                ok, detail = _send_telegram(args.get("message", ""))
   171→                messages.append(msg)
   172→                messages.append({
   173→                    "role":         "tool",
   174→                    "tool_call_id": getattr(tc, "id", ""),
   175→                    "content":      "Enviado no Telegram." if ok else f"Falha: {detail}",
   176→                })
   177→                return _chat_with_tools(client, messages)
   178→    return msg.content or ""

   188→    # Aceita texto direto (mais rápido, sem Whisper) ou arquivo de áudio

   198→                if not audio_bytes:

   200→                file_like      = io.BytesIO(audio_bytes)
   201→                file_like.name = audio_file.filename or "audio.webm"
   202→                transcript     = client.audio.transcriptions.create(
   203→                    model="whisper-1", file=file_like, language="pt"
   204→                )
   205→                user_text = (transcript.text or "").strip()
   206→            except Exception as e:
   207→                return jsonify({"error": f"Transcrição: {e}"}), 500
   208→
   209→    if not user_text:
   210→        return jsonify({"error": "Envie 'text' (JSON) ou arquivo 'audio'"}), 400
   211→
   212→    try:
   213→        messages      = [{"role": "system", "content": SYSTEM_PROMPT},
   214→                         {"role": "user",   "content": user_text}]
   215→        response_text = _chat_with_tools(client, messages).strip()
   216→        if not response_text:
   217→            return jsonify({"error": "Resposta vazia"}), 500
   218→
   219→        tts = client.audio.speech.create(model="tts-1", voice="onyx", input=response_text)
   220→        audio_out = (getattr(tts, "content", None)
   221→                     or (b"".join(tts.iter_bytes()) if hasattr(tts, "iter_bytes") else b""))
   222→        return jsonify({
   223→            "text":        response_text,
   224→            "audio":       base64.b64encode(audio_out).decode() if audio_out else "",
   225→            "transcribed": user_text,
   226→        })
   227→    except Exception as e:
   228→        return jsonify({"error": str(e)}), 500

   713→def _parse_creative_variants(raw: str) -> list[dict]:
   714→    """
   715→    Tenta extrair uma lista de {headline, subheadline} a partir de um JSON retornado pela IA.
   716→    """
   717→    try:
   718→        raw = raw.strip()
   719→        start = raw.find("[")
   720→        end = raw.rfind("]") + 1
   721→        if start == -1 or end == 0:
   722→            raise ValueError("JSON não encontrado na resposta")
   723→        slice_ = raw[start:end]
   724→        parsed = json.loads(slice_)
   725→        if not isinstance(parsed, list):
   726→            raise ValueError("JSON não é uma lista")
   727→        variants: list[dict] = []
   728→        for item in parsed[:5]:
   729→            hl = str(item.get("headline", "")).strip()
   730→            sh = str(item.get("subheadline", "")).strip()
   731→            if hl:
   732→                variants.append({"headline": hl, "subheadline": sh})
   733→        if not variants:
   734→            raise ValueError("Nenhuma variação válida encontrada")
   735→        while len(variants) < 5:
   736→            variants.append(variants[-1])
   737→        return variants
   738→    except Exception as exc:  # pragma: no cover - diagnóstico
   739→        print("[Jake] Erro ao parsear variantes de criativos:", exc, raw[:400])
   740→        raise

   974→def _generate_flux(prompt: str, token: str, style_suffix: str | None = None) -> str:
   975→    """Generate image with Flux 1.1 Pro via Replicate. Returns image URL."""
   976→    import time as _time
   977→    final_prompt = f"{prompt}. {_IMG_MASTER_STYLE}"
   978→    if style_suffix:
   979→        final_prompt = f"{final_prompt}. {style_suffix}"
   980→    resp = requests.post(
   981→        "https://api.replicate.com/v1/models/black-forest-labs/flux-1.1-pro/predictions",
   982→        headers={
   983→            "Authorization": f"Bearer {token}",
   984→            "Content-Type": "application/json",
   985→            "Prefer": "wait=60",
   986→        },
   987→        json={
   988→            "input": {
   989→                "prompt": final_prompt,
   990→                "aspect_ratio": "4:5",
   991→                "output_format": "webp",
   992→                "output_quality": 92,
   993→                "safety_tolerance": 2,
   994→                "prompt_upsampling": True,
   995→            }
   996→        },
   997→        timeout=75,
   998→    )
   999→    if not resp.ok:
  1000→        detail = ""
  1001→        try:
  1002→            data = resp.json()
  1003→            detail = str(data.get("detail") or data.get("error") or "")[:200]
  1004→        except Exception:
  1005→            detail = resp.text[:200]
  1006→        lower_detail = detail.lower()
  1007→        if resp.status_code == 429 or "throttled" in lower_detail or "rate limit" in lower_detail:
  1008→            raise RuntimeError(
  1009→                "Replicate: limite de velocidade atingido. "
  1010→                "Com saldo baixo (< US$5), o limite é de 6 requisições por minuto com burst de 1. "
  1011→                "Gere menos slides por vez (ex.: 1–3) ou espere alguns segundos antes de tentar novamente."
  1012→            )
  1013→        raise RuntimeError(f"Replicate {resp.status_code}: {detail}")
  1014→
  1015→    pred = resp.json()
  1016→    # Synchronous path
  1017→    if pred.get("status") == "succeeded":
  1018→        out = pred.get("output")
  1019→        return (out[0] if isinstance(out, list) else out)
  1020→
  1021→    # Async polling fallback
  1022→    get_url = (pred.get("urls") or {}).get("get")
  1023→    if get_url:
  1024→        for _ in range(25):
  1025→            _time.sleep(3)

  1027→            if p.get("status") == "succeeded":
  1028→                out = p.get("output")
  1029→                return (out[0] if isinstance(out, list) else out)
  1030→            if p.get("status") in ("failed", "canceled"):

  1032→    raise RuntimeError("Flux: timeout após polling.")
