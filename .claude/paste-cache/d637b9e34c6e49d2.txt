const SYSTEM_PROMPT = `Você é um Engenheiro de Prompts Sênior com mais de 20 anos de experiência criando prompts estruturados de alta performance para os mais diversos contextos: marketing, tecnologia, educação, jurídico, criativo, negócios e muito mais.

---

**ETAPA 1 — PERGUNTAS ESTRATÉGICAS**

Quando o usuário apresentar uma ideia ou projeto, você NUNCA gera o prompt direto. Primeiro, você faz de 5 a 7 perguntas estratégicas e objetivas para entender:
- O objetivo principal do prompt
- O público-alvo ou destinatário
- O contexto de uso (plataforma, ferramenta, situação)
- Tom e linguagem desejados
- Restrições ou requisitos específicos
- Exemplos de resultados esperados (se houver)

Formate as perguntas assim (use exatamente este formato JSON para eu processar):
{"type":"questions","questions":["Pergunta 1 aqui?","Pergunta 2 aqui?","Pergunta 3 aqui?","Pergunta 4 aqui?","Pergunta 5 aqui?","Pergunta 6 aqui (se necessário)?","Pergunta 7 aqui (se necessário)?"]}

---

**ETAPA 2 — GERAÇÃO DO PROMPT ESTRUTURADO**

Após o usuário responder suas perguntas, gere o prompt final. Use esta estrutura:

{"type":"prompt","title":"Título descritivo do prompt","prompt":"O prompt completo e estruturado aqui, com toda a riqueza de detalhes, instruções claras, persona se aplicável, contexto, formato de saída esperado, restrições, exemplos se relevante, e tudo que torna o prompt verdadeiramente assertivo e profissional."}

---

**REGRAS IMPORTANTES:**
- Responda SEMPRE em português brasileiro
- Seja direto e profissional
- Nas perguntas, seja estratégico — cada pergunta deve extrair informação que muda significativamente o resultado final
- No prompt final, seja generoso em detalhes: um bom prompt é rico, estruturado e não deixa margem para ambiguidade
- Se a resposta do usuário for insuficiente para gerar o prompt, faça perguntas de refinamento
- Fora dos JSONs, pode conversar normalmente com o usuário`;

let conversationHistory = [];
let isWaiting = false;

function autoResize(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 160) + 'px';
}

function handleKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendMessage();
  }
}

function scrollToBottom() {
  const area = document.getElementById('chatArea');
  setTimeout(() => area.scrollTop = area.scrollHeight, 50);
}

function addMessage(role, content) {
  const area = document.getElementById('chatArea');
  const msgDiv = document.createElement('div');
  msgDiv.className = `msg ${role}`;

  const avatarDiv = document.createElement('div');
  avatarDiv.className = 'msg-avatar';
  avatarDiv.textContent = role === 'agent' ? '🧠' : '👤';

  const bubbleDiv = document.createElement('div');
  bubbleDiv.className = 'bubble';

  // Try to parse JSON from agent
  if (role === 'agent') {
    const rendered = renderAgentContent(content);
    bubbleDiv.innerHTML = rendered;
  } else {
    bubbleDiv.textContent = content;
  }

ild(msgDiv);
  scrollToBottom();
}

function renderAgentContent(text) {
  // Try to find JSON blocks in the text
  const jsonMatch = text.match(/\{[\s\S]*"type"\s*:\s*"(questions|prompt)"[\s\S]*\}/);
  
  if (jsonMatch) {
    try {
      const jsonStr = jsonMatch[0];
      const data = JSON.parse(jsonStr);
      
      // Get text before and after JSON
      const beforeJson = text.substring(0, jsonMatch.index).trim();
      const afterJson = text.substring(jsonMatch.index + jsonStr.length).trim();
      
      let html = '';
      if (beforeJson) html += `<p>${formatText(beforeJson)}</p>`;
      
      if (data.type === 'questions') {
        html += `
          <p style="margin-top:${beforeJson ? '12px' : '0'}">Para criar o prompt mais assertivo possível, preciso entender melhor o projeto. Responda as perguntas abaixo:</p>
          <div class="questions-list">
            ${data.questions.map((q, i) => `
              <div class="question-item">
                <span>${i + 1}.</span>${q}
              </div>
            `).join('')}
          </div>
        `;
      } else if (data.type === 'prompt') {
        html += `
          <p style="margin-top:${beforeJson ? '12px' : '0'}">✅ <strong>Prompt estruturado gerado com sucesso!</strong> Aqui está:</p>
          <div class="prompt-box">
            <div class="prompt-box-header">
              <span>📋 ${data.title || 'PROMPT FINAL'}</span>
              <button class="copy-btn" onclick="copyPrompt(this, \`${escapeTemplate(data.prompt)}\`)">Copiar</button>
            </div>
            ${data.prompt.split('\n').map(line => `<div>${formatText(line) || '&nbsp;'}</div>`).join('')}
          </div>
        `;
      }
      
      if (afterJson) html += `<p style="margin-top:12px">${formatText(afterJson)}</p>`;
      return html;
    } catch(e) {
      // fallback
    }
  }
  
  return formatText(text);
}

function formatText(text) {
  return text
    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
    .replace(/\n/g, '<br>');
}

function escapeTemplate(str) {
  return str.replace(/`/g, '\\`').replace(/\$/g, '\\$');
}

function copyPrompt(btn, text) {
  navigator.clipboard.writeText(text).then(() => {
    btn.textContent = '✓ Copiado!';
    setTimeout(() => btn.textContent = 'Copiar', 2000);
  });
}

function addTyping() {
  const area = document.getElementById('chatArea');
  const typingDiv = document.createElement('div');
  typingDiv.className = 'msg agent';
  typingDiv.id = 'typingIndicator';
  typingDiv.innerHTML = `
    <div class="msg-avatar">🧠</div>
    <div class="bubble">
      <div class="typing"><span></span><span></span><span></span></div>
    </div>
  `;
  area.appendChild(typingDiv);
  scrollToBottom();
}

function removeTyping() {
  const el = document.getElementById('typingIndicator');
  if (el) el.remove();
}

async function sendMessage() {
  if (isWaiting) return;
  const input = document.getElementById('userInput');
  const text = input.value.trim();
  if (!text) return;

  input.value = '';
  input.style.height = 'auto';
  isWaiting = true;
  document.getElementById('sendBtn').disabled = true;

  addMessage('user', text);
  conversationHistory.push({ role: 'user', content: text });

  addTyping();

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: SYSTEM_PROMPT,
        messages: conversationHistory
      })
    });

    const data = await response.json();
    const reply = data.content?.map(b => b.text || '').join('') || 'Erro ao obter resposta.';

    removeTyping();
    conversationHistory.push({ role: 'assistant', content: reply });
    addMessage('agent', reply);

  } catch (err) {
    removeTyping();
    addMessage('agent', '⚠️ Erro de conexão. Verifique sua rede e tente novamente.');
  }

  isWaiting = false;
  document.getElementById('sendBtn').disabled = false;
  scrollToBottom();
}

  `;
}
